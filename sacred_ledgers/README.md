function recordTransaction(transaction, ledgerPath) {

  // Generate transaction ID
  const txId = `TX-${Date.now()}-${crypto.randomBytes(4).toString('hex')}`;

  // Save ledger
  fs.writeFileSync(ledgerPath, JSON.stringify(ledger, null, 2));
  
  console.log(`âœ… Transaction recorded: ${txId}`);

function generateSealHash(data) {
  return crypto
    .createHash('sha256')
    .update(JSON.stringify(data))
    .digest('hex');
