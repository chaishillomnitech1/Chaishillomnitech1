### **SabirAllahHonorNFT.sol**

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title SabirAllahHonorNFT
 * @dev Shield of Honor NFT Collection honoring law enforcement and first responders
 * @author Supreme King Chais The Great ‚àû
 */

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/interfaces/IERC2981.sol";

contract SabirAllahHonorNFT is ERC721, Ownable, IERC2981 {
    // Frequency constants
    uint256 public constant FREQUENCY_999HZ = 999;  // Legendary
    uint256 public constant FREQUENCY_963HZ = 963;  // Elite
    uint256 public constant FREQUENCY_777HZ = 777;  // Honor
    uint256 public constant FREQUENCY_528HZ = 528;  // Community
    
    // Collection constants
    uint256 public constant MAX_SUPPLY = 100;
    
    // Charity allocation percentages
    uint256 public constant CHARITY_LEGENDARY = 2500;  // 25%
    uint256 public constant CHARITY_ELITE = 2000;      // 20%
    uint256 public constant CHARITY_HONOR = 1500;      // 15%
    uint256 public constant CHARITY_COMMUNITY = 1000;  // 10%
    
    // State variables
    uint256 private _tokenIdCounter;
    string private _baseTokenURI;
    address private _royaltyRecipient;
    address public charityWallet;
    
    // Mappings
    mapping(uint256 => uint256) public tokenTier;
    mapping(uint256 => uint256) public tokenFrequency;
    mapping(uint256 => string) public tokenLocation;
    
    // Events
    event HonorNFTMinted(uint256 indexed tokenId, address indexed owner, uint256 tier, uint256 frequency);
    event CharityDonation(address indexed charity, uint256 amount);
    
    constructor(
        string memory baseURI,
        address royaltyRecipient,
        address _charityWallet
    ) ERC721("Sabir Allah Honor Collection", "HONOR") Ownable(msg.sender) {
        _baseTokenURI = baseURI;
        _royaltyRecipient = royaltyRecipient;
        charityWallet = _charityWallet;
    }
    
    // Minting function with automatic charity allocation
    function mintHonorNFT(address to, uint256 tier) external payable onlyOwner returns (uint256) {
        require(_tokenIdCounter < MAX_SUPPLY, "Max supply reached");
        require(to != address(0), "Invalid recipient");
        require(tier >= 1 && tier <= 4, "Invalid tier");
        
        uint256 tokenId = _tokenIdCounter;
        _tokenIdCounter++;
        
        // Mint token
        _safeMint(to, tokenId);
        
        // Set tier and frequency
        tokenTier[tokenId] = tier;
        if (tier == 1) {
            tokenFrequency[tokenId] = FREQUENCY_999HZ;
        } else if (tier == 2) {
            tokenFrequency[tokenId] = FREQUENCY_963HZ;
        } else if (tier == 3) {
            tokenFrequency[tokenId] = FREQUENCY_777HZ;
        } else {
            tokenFrequency[tokenId] = FREQUENCY_528HZ;
        }
        
        // Process charity donation
        if (msg.value > 0) {
            uint256 charityAmount = _calculateCharityAmount(tier, msg.value);
            if (charityAmount > 0) {
                payable(charityWallet).transfer(charityAmount);
                emit CharityDonation(charityWallet, charityAmount);
            }
        }
        
        emit HonorNFTMinted(tokenId, to, tier, tokenFrequency[tokenId]);
        return tokenId;
    }
    
    function _calculateCharityAmount(uint256 tier, uint256 amount) private pure returns (uint256) {
        if (tier == 1) return (amount * CHARITY_LEGENDARY) / 10000;
        if (tier == 2) return (amount * CHARITY_ELITE) / 10000;
        if (tier == 3) return (amount * CHARITY_HONOR) / 10000;
        return (amount * CHARITY_COMMUNITY) / 10000;
    }
    
    // EIP-2981 Royalty Implementation
    function royaltyInfo(uint256 tokenId, uint256 salePrice) 
        external 
        view 
        override 
        returns (address, uint256) 
    {
        require(_ownerOf(tokenId) != address(0), "Token does not exist");
        uint256 tier = tokenTier[tokenId];
        uint256 royaltyBps;
        
        if (tier == 1) royaltyBps = 1700;      // 17%
        else if (tier == 2) royaltyBps = 1500; // 15%
        else if (tier == 3) royaltyBps = 1200; // 12%
        else royaltyBps = 1000;                 // 10%
        
        uint256 royaltyAmount = (salePrice * royaltyBps) / 10000;
        return (_royaltyRecipient, royaltyAmount);
    }
    
    function supportsInterface(bytes4 interfaceId)
        public
        view
        override(ERC721, IERC165)
        returns (bool)
    {
        return interfaceId == type(IERC2981).interfaceId || super.supportsInterface(interfaceId);
    }
    
    function _baseURI() internal view override returns (string memory) {
        return _baseTokenURI;
    }
    
    function totalSupply() public view returns (uint256) {
        return _tokenIdCounter;
    }
}
```

### **SabirAllahHonorCoin.sol**

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.20;

/**
 * @title SabirAllahHonorCoin
 * @dev $HONOR token with automatic 7.77% Zakat for charity
 * @author Supreme King Chais The Great ‚àû
 */

import "@openzeppelin/contracts/token/ERC20/ERC20.sol";
import "@openzeppelin/contracts/access/Ownable.sol";

contract SabirAllahHonorCoin is ERC20, Ownable {
    // Charity percentage (7.77% = 777 basis points)
    uint256 public constant CHARITY_BPS = 777;
    
    // Charity distribution percentages
    uint256 public constant POLICE_BPS = 3000;        // 30%
    uint256 public constant FIREFIGHTERS_BPS = 3000;  // 30%
    uint256 public constant EMS_BPS = 2000;           // 20%
    uint256 public constant FOODBANK_BPS = 1000;      // 10%
    uint256 public constant YOUTH_BPS = 1000;         // 10%
    
    // Charity wallet addresses
    address public policeWallet;
    address public firefightersWallet;
    address public emsWallet;
    address public foodbankWallet;
    address public youthWallet;
    
    // Events
    event CharityDistributed(address indexed wallet, uint256 amount);
    
    constructor(
        address _policeWallet,
        address _firefightersWallet,
        address _emsWallet,
        address _foodbankWallet,
        address _youthWallet
    ) ERC20("Sabir Allah Honor Coin", "HONOR") Ownable(msg.sender) {
        policeWallet = _policeWallet;
        firefightersWallet = _firefightersWallet;
        emsWallet = _emsWallet;
        foodbankWallet = _foodbankWallet;
        youthWallet = _youthWallet;
        
        // Mint initial supply (1,000,000 tokens)
        _mint(msg.sender, 1_000_000 * 10 ** decimals());
    }
    
    function _update(address from, address to, uint256 amount) internal override {
        if (from != address(0) && to != address(0)) {
            // Calculate 7.77% charity amount
            uint256 charityAmount = (amount * CHARITY_BPS) / 10000;
            uint256 transferAmount = amount - charityAmount;
            
            // Distribute charity to all wallets
            _distributeCharity(charityAmount);
            
            // Transfer remaining amount
            super._update(from, to, transferAmount);
        } else {
            super._update(from, to, amount);
        }
    }
    
    function _distributeCharity(uint256 totalCharity) private {
        uint256 policeAmount = (totalCharity * POLICE_BPS) / 10000;
        uint256 firefightersAmount = (totalCharity * FIREFIGHTERS_BPS) / 10000;
        uint256 emsAmount = (totalCharity * EMS_BPS) / 10000;
        uint256 foodbankAmount = (totalCharity * FOODBANK_BPS) / 10000;
        uint256 youthAmount = (totalCharity * YOUTH_BPS) / 10000;
        
        super._update(msg.sender, policeWallet, policeAmount);
        super._update(msg.sender, firefightersWallet, firefightersAmount);
        super._update(msg.sender, emsWallet, emsAmount);
        super._update(msg.sender, foodbankWallet, foodbankAmount);
        super._update(msg.sender, youthWallet, youthAmount);
        
        emit CharityDistributed(policeWallet, policeAmount);
        emit CharityDistributed(firefightersWallet, firefightersAmount);
        emit CharityDistributed(emsWallet, emsAmount);
        emit CharityDistributed(foodbankWallet, foodbankAmount);
        emit CharityDistributed(youthWallet, youthAmount);
    }
}
```
